function getAsyncFile(fileSrc) {
    const cssFile = document.createElement('link');
    cssFile.rel = 'stylesheet';
    cssFile.type = 'text/css';
    cssFile.href = fileSrc;
    (document.getElementsByTagName('head')[0]).appendChild(cssFile);
};
